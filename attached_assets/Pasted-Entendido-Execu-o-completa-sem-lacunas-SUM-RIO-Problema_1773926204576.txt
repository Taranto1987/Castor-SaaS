## 1. TRANSAÇÃO (OBRIGATÓRIO)

```javascript
// /api/fechar-venda.js

await db.transaction(async trx => {

  const orcamento = await trx('orcamentos')
    .where({ id: orcamento_id })
    .first();

  if (orcamento.status === 'VENDIDO') return;

  const venda_id = uuid();

  await trx('vendas').insert({
    id: venda_id,
    orcamento_id,
    vendedor_id: orcamento.vendedor_id,
    valor: orcamento.valor
  });

  await trx('orcamentos')
    .update({ status: 'VENDIDO' })
    .where({ id: orcamento_id });

  await trx('entregas').insert({
    id: uuid(),
    venda_id,
    endereco: orcamento.endereco,
    status: 'PENDENTE'
  });

});
```

import bcrypt from 'bcrypt';

export async function hashSenha(senha) {
  return await bcrypt.hash(senha, 10);
}

export async function verificarSenha(senha, hash) {

}

// RBAC

function authorize(role) {
  return (req, res, next) => {
    if (req.user.role !== role) return res.status(403).end();
    next();
  };
}

## 3. MÁQUINA DE ESTADOS

```javascript
// /engine/status.js

export const STATUS = {
  ORCAMENTO: ['PENDENTE','APROVADO','VENDIDO'],
  ENTREGA: ['PENDENTE','ROTEIRIZADO','EM_ROTA','ENTREGUE']
};

export function validarTransicao(atual, proximo, tipo) {
  const fluxo = STATUS[tipo];
  const i = fluxo.indexOf(atual);
  if (fluxo[i + 1] !== proximo) {
    throw new Error("TRANSICAO_INVALIDA");
  }
}
```
