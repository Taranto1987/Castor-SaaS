import PDFDocument from "pdfkit"
import fs from "fs"

export function gerarPDF(cliente,produtos,totais){

const doc = new PDFDocument()

doc.pipe(fs.createWriteStream(`orcamento-${Date.now()}.pdf`))

doc.fontSize(18).text("CASTOR CABO FRIO")

doc.moveDown()

doc.text(`Cliente: ${cliente}`)

produtos.forEach((p,i)=>{

doc.text(`${i+1}. ${p.nome}`)

doc.text(`Pix: ${p.preco_pix}`)

doc.text(`Prazo: ${p.preco_prazo}`)

doc.moveDown()

})

doc.text(`Total Pix: ${totais.pix}`)

doc.text(`Total Prazo: ${totais.prazo}`)

doc.end()

}
