# 2 — Carrinho de produtos

Arquivo:

```
components/ProductCart.tsx
```

```tsx
import { useState } from "react"

export default function ProductCart(){

const [items,setItems] = useState([])

function addProduct(produto){

setItems([...items,produto])

}

function removeProduct(index){

const copy = [...items]
copy.splice(index,1)

setItems(copy)

}

return {items,addProduct,removeProduct}

}
```

# 3 — Motor de cálculo

Arquivo:

```
utils/budgetEngine.ts
```

```ts
export function calcular(produtos){

let pix = 0
let prazo = 0

produtos.forEach(p=>{

pix += p.preco_pix
prazo += p.preco_prazo

})

const parcela = prazo / 12

return {

pix,
prazo,
parcela

}

}
```

# 4 — Template VIP fixo

Arquivo:

```
utils/vipTemplate.ts
```

```ts
import { calcular } from "./budgetEngine"

export function gerarVIP(cliente,produtos,obs){

const calc = calcular(produtos)

const lista = produtos.map((p,i)=>{

return `${i+1}️⃣ ${p.nome}
Altura: ${p.altura}cm
Ref: ${p.sku}
Pix: R$ ${p.preco_pix}
Prazo: R$ ${p.preco_prazo}`

}).join("\n\n")

return `━━━━━━━━━━━━━━━━━━━━━━
🇧🇷 CASTOR CABO FRIO
━━━━━━━━━━━━━━━━━━━━━━

Cliente: ${cliente}

Produtos:
${lista}

━━━━━━━━━━━━━━━━━━━━━━

💰 PIX
R$ ${calc.pix.toFixed(2)}

💳 Parcelado
R$ ${calc.prazo.toFixed(2)}

12x de R$ ${calc.parcela.toFixed(2)}

━━━━━━━━━━━━━━━━━━━━━━

Obs:
${obs}

📞 (22) 99241-0112
`

}
```
